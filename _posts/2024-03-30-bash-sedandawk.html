bash sed awk几种匹配方法

$ head oracle2.txt|tr '$\r\n' ' '|sed -e 's/huawei/\r\nhuawei/g'|awk -F'|' '{print $38}'

head oracle3.txt|tr '$\r\n' ' '|sed -e 's/ALL_ROWS/\r\nALL_ROWS/g'|awk -F'","' '{print $58}'

head mysql.txt |sed -e 's/^[0-9]\{4\}-[0-2]\{2\}-[0-9]\{2\}T/mysqlstart/g'|tr '$\r\n' ' '| sed -e 's/mysqlstart/\r\nmysqlstart/g'|awk '{for(i=3;i<=NF;i++)printf("%s ",$i);print ""}'

$ cat aaaa|sed -e 's/|||/+++/g'|awk -F"+++" '{print $21}'
